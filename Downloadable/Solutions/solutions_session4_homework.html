## Animal base class
class Animal:  ## class definition
    def __init__(self, name, breed, sound):
        self.name = name
        self.breed = breed
        self.sound = sound

    def speak(self):
        return f"{self.name} says {self.sound}!"
In[2]:
## Create `Speech` mixin
class Speech:  ## mixin that adds behaviors to animal class
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.sound_list = [self.sound]  ## create an empty sound list.

    def learn_sound(self, sound):  ## use this as a base to create your 'learn' method
        if sound not in self.sound_list:
            self.sound_list.append(sound)
            return f"{self.name} learned a new sound: {sound}!"
        return f"{self.name} already knows {sound}!"

    def set_sound(self, sound_number):
        if 0 <= sound_number < len(self.sound_list):
            self.sound = self.sound_list[sound_number]
            return f"Set sound to: {self.sound}!"
        else:
            return (
                f"Invalid input. Must be a number between 0 and {len(self.sound_list)}."
            )

## Create Parrot Class:
class Parrot(Animal, Speech):  ## inherit from Speech and Animal
    def __init__(
        self, name, breed="parrot", sound="squawk"
    ):  ## add arguments here as needed
        super().__init__(name, breed, sound)  ## initialize Animal
        Speech.__init__(self)  ## initialize Speech

    def __str__(self):
        return f"Name: {self.name}. Current sound: {self.sound}"

    def __repr__(self):
        return f"Name: {self.name}. Current sound: {self.sound} \n Known sounds: {self.sound_list}"

Z = linkage(X_scaled, method="ward")
